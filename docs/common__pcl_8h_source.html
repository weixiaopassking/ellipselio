27  * @brief EllipseLio unified point type with geometry, color, intensity, and
28  * metadata.
29  * @details Combines position (x,y,z), RGBA color, intensity, timestamp,

32  */
33 struct EIGEN_ALIGN16 PointXYZNRGBIT {

35  PCL_ADD_POINT4D;

37  union {
38  struct {
39  /// @brief Range bin index for octree organization
40  uint32_t bin_idx;
41  /// @brief Scan line index in point cloud
42  uint32_t scan_idx;
43  /// @brief Flag indicating whether RGBA color is valid
44  uint32_t has_rgb;
45  /// @brief Primitive type (0=point, 1=line, 2=plane, 3=ellipsoid)
46  uint32_t prim_type;
47  };
48  /// @brief Padding array for data alignment
49  float data_n[4];
50  };
51  /// @brief Color and intensity data
52  union {

54  /// @brief RGBA color (OpenCV convention: BGR packed in 32-bit int)
55  PCL_ADD_UNION_RGB

57  float intensity;

59  uint32_t time_secs;
60  /// @brief Timestamp nanoseconds component
61  uint32_t time_nsecs;
62  };
63  /// @brief Padding array for data alignment
64  float data_c[4];
65  };
66  /// @brief SIMD-aligned color access
67  PCL_ADD_EIGEN_MAPS_RGB
