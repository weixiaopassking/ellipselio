56 void ImuProcess::Process(const sensor_msgs::msg::Imu::SharedPtr msg) {
57  double dt;
58  input_ikfom in;
59  ImuState imu_state;
60  V3D gyr_avr, acc_avr;
61  rclcpp::Time msg_time;
62 
63  if (imu_need_init_) {
64  InitImu(msg);
65  return;
66  }
67 
68  if (!imu_states_.size()) {
69  acc_avr = mean_acc_ * kGravityMetersPerSecondSquared / mean_acc_.norm();
70  gyr_avr = mean_gyr_;
71  } else {
72  gyr_avr << msg->angular_velocity.x, msg->angular_velocity.y,
73  msg->angular_velocity.z;
74  gyr_avr += imu_states_.back().gyr;
75  gyr_avr *= 0.5;
76  acc_avr << msg->linear_acceleration.x, msg->linear_acceleration.y,
77  msg->linear_acceleration.z;
78  acc_avr += imu_states_.back().acc;
79  acc_avr *= 0.5 * kGravityMetersPerSecondSquared / mean_acc_.norm();
80  }
81 
82  msg_time = msg->header.stamp;
83  dt = (msg_time - kf_state_.time).seconds();
84 
85  in.acc = acc_avr;
86  in.gyro = gyr_avr;
87  kf_->predict(dt, q_, in);
88 
89  kf_state_.state = kf_->get_x();
90  kf_state_.cov = kf_->get_P();
91  kf_state_.time = msg_time;
92  kf_state_.gyr << msg->angular_velocity.x, msg->angular_velocity.y,
93  msg->angular_velocity.z;
94 
95  SetKfState();
96 
97  imu_state.state = kf_state_;
98  imu_state.acc << msg->linear_acceleration.x, msg->linear_acceleration.y,
99  msg->linear_acceleration.z;
100  imu_state.gyr << msg->angular_velocity.x, msg->angular_velocity.y,
101  msg->angular_velocity.z;
102  imu_state.acc_avr = kf_state_.state.rot * (acc_avr - kf_state_.state.ba);
103  imu_state.acc_avr += kf_state_.state.grav.get_vect();
104  imu_state.gyr_avr = gyr_avr - kf_state_.state.bg;
105 
106  imu_states_.push_back(imu_state);
107  imu_start_time_ = imu_states_.front().state.time;
108  imu_end_time_ = imu_states_.back().state.time;
109  imu_has_data_ = true;
110 }

139

143  kf_state_.state.bg = mean_gyr_;

163 void ImuProcess::GetTimeMatch(
164  int* match_idx, const rclcpp::Time& match_time,
165  const boost::circular_buffer<ImuState>& imu_states) {
166  double time_diff;
167  bool match_flag = false;
168 
169  time_diff = (match_time - imu_states.front().state.time).seconds();
170  *match_idx = std::floor(time_diff * params_.rate);
171  *match_idx = std::max(*match_idx, 0);
172  *match_idx = std::min(*match_idx, static_cast<int>(imu_states.size()) - 1);
173 
174  while (!match_flag) {
175  if (imu_states[*match_idx].state.time > match_time) {
176  if (*match_idx == 0) {
177  match_flag = true;
178  } else if (imu_states[*match_idx - 1].state.time > match_time) {
179  (*match_idx)--;
180  } else {
181  match_flag = true;
182  }
183  } else if (*match_idx == imu_states.size() - 1) {
184  match_flag = true;
185  } else {
186  (*match_idx)++;
187  }
188  }

191 void ImuProcess::UndistortPointCloud(EllipseLioPointCloudPtr pc,
192  KfState* kf_state,
193  const rclcpp::Time& lidar_start_time,
194  const rclcpp::Time& lidar_end_time,
195  const CamProcessVec& cams) {
196  int match_idx;
197  Eigen::Isometry3d T_imu_lidar, T_world_imu_e;
198 
199  GetMatchingImages(lidar_start_time, lidar_end_time, cams, synced_imu_states_);
200  GetTimeMatch(&match_idx, lidar_end_time, synced_imu_states_);
201 
202  *kf_state = synced_imu_states_[match_idx].state;
203 
204  T_imu_lidar.linear() =
205  synced_imu_states_[match_idx].state.state.offset_R_L_I.toRotationMatrix();
206  T_imu_lidar.translation() =
207  synced_imu_states_[match_idx].state.state.offset_T_L_I;
208  T_world_imu_e.linear() =
209  synced_imu_states_[match_idx].state.state.rot.toRotationMatrix();
210  T_world_imu_e.translation() = synced_imu_states_[match_idx].state.state.pos;
211 
212 #pragma omp parallel for
213  for (size_t i = 0; i < pc->points.size(); i++) {
214  int head_idx, tail_idx;
215  Eigen::Isometry3d T_world_imu_p, T_imu_e_imu_p;
216 
217  rclcpp::Time pt_time = rclcpp::Time(pc->points[i].time_secs,
218  pc->points[i].time_nsecs, RCL_ROS_TIME);
219 
220  GetTimeMatch(&tail_idx, pt_time, synced_imu_states_);
221  head_idx = std::max(tail_idx - 1, 0);
222 
223  M3D R_imu = synced_imu_states_[head_idx].state.state.rot.toRotationMatrix();
224  V3D vel_imu = synced_imu_states_[head_idx].state.state.vel;
225  V3D pos_imu = synced_imu_states_[head_idx].state.state.pos;
226  V3D acc_avr = synced_imu_states_[tail_idx].acc_avr;
227  V3D gyr_avr = synced_imu_states_[tail_idx].gyr_avr;
228 
229  double dt = (pt_time - synced_imu_states_[head_idx].state.time).seconds();
230 
231  T_world_imu_p.linear() = R_imu * Exp(gyr_avr, dt);
232  T_world_imu_p.translation() =
233  pos_imu + vel_imu * dt + 0.5 * acc_avr * dt * dt;
234  T_imu_e_imu_p = T_world_imu_e.inverse() * T_world_imu_p;
235 
236  ColorisePoint(&pc->points[i], cams, T_world_imu_p, T_imu_lidar);
237 
238  pc->points[i].getVector3fMap() =
239  (T_imu_lidar.inverse() * T_imu_e_imu_p * T_imu_lidar *
240  pc->points[i].getVector3fMap().cast<double>())
241  .cast<float>();
242  }
243 }

277 void ImuProcess::ColorisePoint(EllipseLioPoint* pt, const CamProcessVec& cams,
278  const Eigen::Isometry3d& T_world_pt,
279  const Eigen::Isometry3d& T_imu_lidar) {
280  int valid_num;
281  Eigen::MatrixXi cam_cols;
282  Eigen::Vector3i cam_col;
283  Eigen::VectorXi cam_sum;
284 
285  pt->r = 0;
286  pt->g = 0;
287  pt->b = 0;
288  pt->a = 0;
289  pt->has_rgb = false;
290 
291  if (!cams.size()) return;
292  cam_sum = Eigen::VectorXi::Zero(cams.size());
293  cam_cols = Eigen::MatrixXi::Zero(cams.size(), 3);
294 
295 #pragma omp parallel for
296  for (size_t i = 0; i < cams.size(); i++) {
297  Eigen::Vector3i pt_col;
298  Eigen::Vector3d pt_img;
299 
300  if (!cams[i]->has_img_match_) continue;
301 
302  Eigen::Isometry3d& T_cam_lidar = cams[i]->T_cam_lidar_;
303  Eigen::Isometry3d& T_world_img = cams[i]->T_world_img_;
304 
305  pt_col = Eigen::Vector3i::Zero();
306  pt_img = T_cam_lidar * T_imu_lidar.inverse() * T_world_img.inverse() *
307  T_world_pt * T_imu_lidar * pt->getVector3fMap().cast<double>();
308 
309  if (cams[i]->ColorPoint(&pt_img, &pt_col)) {
310  cam_cols.row(i) = pt_col;
311  cam_sum(i) = 1;
312  }
313  }
314 
315  cam_col = cam_cols.colwise().sum();
316  valid_num = cam_sum.sum();
317 
318  if (valid_num) {
319  cam_col /= valid_num;
320 
321  pt->a = 255;
322  pt->r = cam_col(0);
323  pt->g = cam_col(1);
324  pt->b = cam_col(2);
325  pt->has_rgb = true;
326  }
