110 void LidarProcess::Process(const sensor_msgs::msg::PointCloud2::SharedPtr msg) {
111  auto& clk = *node_->get_clock();
112 
113  switch (static_cast<LidType>(params_.type)) {
114  case LidType::kLivox:
115  PointCloudHandler<LivoxPoint>(msg);
116  break;
117  case LidType::kVelodyne:
118  PointCloudHandler<VelodynePoint>(msg);
119  break;
120  case LidType::kOuster:
121  PointCloudHandler<OusterPoint>(msg);
122  break;
123  case LidType::kHesai:
124  PointCloudHandler<HesaiPoint>(msg);
125  break;
126  case LidType::kGazebo:
127  PointCloudHandler<GazeboPoint>(msg);
128  break;
129  }
130 
131  int end_bin = num_bins_;
132  int prev_start_bin = -1;
133  while (start_bin_ > prev_start_bin && prev_start_bin < max_octree_res_) {
134 #pragma omp parallel for
135  for (size_t i = 0; i < end_bin; i++) {
136  std::vector<int> new_idxs, added_idxs;
137  if (!bin_sizes_[i]) continue;
138 
139  bin_pcs_sizes_[i] = 0;
140  bin_pcs_[i].clear();
141  bin_octrees_[i].clear();
142 
143  bin_octrees_[i].SetBucketSize(1);
144  bin_octrees_[i].SetMinExtent(octree_resolutions_[fmax(i, start_bin_)]);
145  bin_octrees_[i].Update(*process_pc_, bin_sizes_[i], bin_idxs_[i],
146  added_idxs, new_idxs);
147 
148  if (!added_idxs.size()) continue;
149  bin_pcs_[i] = EllipseLioPointCloud(*process_pc_, added_idxs);
150  bin_pcs_sizes_[i] = bin_pcs_[i].size();
151  SetMinMaxTime(i);
152 
153  bin_idxs_[i] = added_idxs;
154  bin_sizes_[i] = bin_pcs_sizes_[i];
155  }
156 
157  float mean_bin = 0, mean_num = 0;
158  int proc_points = bin_pcs_sizes_.sum();
159  for (size_t i = 0; i < num_bins_; i++) {
160  if (mean_num > 0.5 * proc_points && bin_sizes_[i] < 0.01 * proc_points) {
161  break;
162  }
163  mean_num += bin_sizes_[i];
164  mean_bin += i * bin_sizes_[i];
165  }
166 
167  prev_start_bin = start_bin_;
168  mean_bin_ = floor(mean_bin / mean_num);
169  start_bin_ = fmin(mean_bin_, max_octree_res_);
170  end_bin = start_bin_;
171  }
172 
173  ellipselio_pc_->clear();
174  bool init_time = true;
175  for (size_t i = 0; i < num_bins_; i++) {
176  if (!bin_pcs_sizes_[i]) continue;
177  if ((ellipselio_pc_->size() + bin_pcs_[i].size()) > kMaxProcPoints) break;
178 
179  *ellipselio_pc_ += bin_pcs_[i];
180 
181  if (init_time) {
182  lidar_start_time_ = bin_min_times_[i];
183  lidar_end_time_ = bin_max_times_[i];
184  init_time = false;
185  } else {
186  lidar_start_time_ = std::min(lidar_start_time_, bin_min_times_[i]);
187  lidar_end_time_ = std::max(lidar_end_time_, bin_max_times_[i]);
188  }
189  }
190  lidar_has_data_ = true;
191 }

303 template <typename InPtType>
304 void LidarProcess::PointCloudHandler(
305  const sensor_msgs::msg::PointCloud2::SharedPtr msg) {
306  pcl::PointCloud<InPtType> in_pc;
307  pcl::fromROSMsg(*msg, in_pc);
308  int num_points, proc_points;
309  std::atomic<int> rej_points = 0;
310  float mean_bin = 0, mean_num = 0;
311 
312  std::fill(bin_sizes_.begin(), bin_sizes_.end(), 0);
313  num_points = fmin(in_pc.size(), kMaxScanPoints);
314 
315 #pragma omp parallel for
316  for (size_t i = 0; i < num_points; i++) {
317  rclcpp::Time point_time = msg->header.stamp;
318  point_time += lidar_time_offset_;
319  ConvertPoint<InPtType>(in_pc, i, point_time);
320  float range = sqrt(process_pc_->points[i].x * process_pc_->points[i].x +
321  process_pc_->points[i].y * process_pc_->points[i].y +
322  process_pc_->points[i].z * process_pc_->points[i].z);
323 
324  if (range < params_.min_range || range > params_.max_range ||
325  std::isnan(range) || std::isinf(range)) {
326  rej_points++;
327  continue;
328  }
329 
330  int bin_idx = floor(range);
331  process_pc_->points[i].bin_idx = bin_idx;
332  bin_idxs_[bin_idx][bin_sizes_[bin_idx]++] = i;
333  }
334 
335  proc_points = num_points - rej_points.load();
336 
337  for (size_t i = 0; i < num_bins_; i++) {
338  if (mean_num > 0.5 * proc_points && bin_sizes_[i] < 0.01 * proc_points) {
339  break;
340  }
341  mean_num += bin_sizes_[i];
342  mean_bin += i * bin_sizes_[i];
343  }
344 
345  mean_bin_ = floor(mean_bin / mean_num);
346  start_bin_ = fmin(mean_bin_, max_octree_res_);
347 }
